# This is a simple example of a macro with a commented line.
# Lines are commented by using a '#' at the start. Valid
# lines in the macro are all variants of /ac that the game
# accepts. For example:
# /ac Ingenuity
# /ac "Ingenuity"
# /ac Ingenuity <wait.2>
# /ac "Ingenuity" <wait.2>
#
# If the wait is omitted then at this time Talan will default
# to a GCD of ~2.75 seconds, even for buffs.
/ac "Inner Quiet" <wait.2>
/ac "Steady Hand ii" <wait.2>
/ac "Prudent Touch" <wait.3>
/ac "Careful Synthesis ii" <wait.3>
